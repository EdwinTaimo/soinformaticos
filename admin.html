function carregarAdmin() { fetch('data.csv') .then(response => response.text()) .then(data => { const rows = data.split('\n'); const tableBody = document.getElementById('table-body'); tableBody.innerHTML = ''; // Clear existing data rows.forEach(row => { const cols = row.split(','); const newRow = document.createElement('tr'); cols.forEach(col => { const newCol = document.createElement('td'); newCol.textContent = col; newRow.appendChild(newCol); }); const actionCol = document.createElement('td'); const button = document.createElement('button'); button.textContent = 'Action'; actionCol.appendChild(button); newRow.appendChild(actionCol); tableBody.appendChild(newRow); }); }) .catch(error => console.error('Error:', error)); }
